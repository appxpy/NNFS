132inline void progressbar::update()
133{
134
135 if (n_cycles == 0)
136 throw std::runtime_error(
137 "progressbar::update: number of cycles not set");
138
139 if (!update_is_called)
140 {
141 if (do_show_bar == true)
142 {
143 output << opening_bracket_char;
144 for (int _ = 0; _ < 50; _++)
145 output << todo_char;
146 output << closing_bracket_char << " 0%";
147 }
148 else
149 output << "0%";
150 }
151 update_is_called = true;
152
153 int perc = 0;
154
155 // compute percentage, if did not change, do nothing and return
156 perc = progress * 100. / (n_cycles - 1);
157 if (perc < last_perc)
158 return;
159
160 // update percentage each unit
161 if (perc == last_perc + 1)
162 {
163 // erase the correct number of characters
164 if (perc <= 10)
165 output << "\b\b" << perc << '%';
166 else if (perc > 10 and perc < 100)
167 output << "\b\b\b" << perc << '%';
168 else if (perc == 100)
169 output << "\b\b\b" << perc << '%';
170 }
171 if (do_show_bar == true)
172 {
173 // update bar every ten units
174 if (perc % 2 == 0)
175 {
176 // erase closing bracket
177 output << std::string(closing_bracket_char.size(), '\b');
178 // erase trailing percentage characters
179 if (perc < 10)
180 output << "\b\b\b";
181 else if (perc >= 10 && perc < 100)
182 output << "\b\b\b\b";
183 else if (perc == 100)
184 output << "\b\b\b\b\b";
185
186 // erase 'todo_char'
187 for (int j = 0; j < 50 - (perc - 1) / 2; ++j)
188 {
189 output << std::string(todo_char.size(), '\b');
190 }
191
192 // add one additional 'done_char'
193 if (perc == 0)
194 output << todo_char;
195 else
196 output << done_char;
197
198 // refill with 'todo_char'
199 for (int j = 0; j < 50 - (perc - 1) / 2 - 1; ++j)
200 output << todo_char;
201
202 // readd trailing percentage characters
203 output << closing_bracket_char << ' ' << perc << '%';
204 }
205 }
206 last_perc = perc;
207 ++progress;
208 output << std::flush;
209
210 return;
211}
