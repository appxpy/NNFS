265 void backward_pass(const Eigen::MatrixXd &labels)
266 {
267 Eigen::MatrixXd da = _loss->gradient(_output, labels);
268
269 for (int i = _num_layers - 1; i >= 0; --i)
270 {
271 auto &layer = std::get<0>(_layers[i]);
272 auto &activation = std::get<1>(_layers[i]);
273
274 Eigen::MatrixXd dz = da.array() * activation->gradient(layer->output()).array();
275
276 Eigen::MatrixXd prev_layer_output;
277 if (i == 0)
278 {
279 prev_layer_output = _input;
280 }
281 else
282 {
283 auto &prev_layer = std::get<0>(_layers[i - 1]);
284 auto &prev_activation = std::get<1>(_layers[i - 1]);
285 prev_layer_output = prev_activation->operator()(prev_layer->output());
286 }
287
288 layer->grad_weights(dz * prev_layer_output.transpose() / _num_examples);
289 layer->grad_weights(layer->grad_weights() +
290 (_regularization_factor / _num_examples) * layer->weights());
291 layer->grad_bias(dz.rowwise().mean());
292 da = layer->weights().transpose() * dz;
293 }
