287 void backward_pass(const Eigen::MatrixXd &labels)
288 {
289 Eigen::MatrixXd da = _loss->gradient(_output, labels);
290
291 for (int i = _num_layers - 1; i >= 0; --i)
292 {
293 auto &layer = std::get<0>(_layers[i]);
294 auto &activation = std::get<1>(_layers[i]);
295
296 Eigen::MatrixXd dz = da.array() * activation->gradient(layer->output()).array();
297
298 Eigen::MatrixXd prev_layer_output;
299 if (i == 0)
300 {
301 prev_layer_output = _input;
302 }
303 else
304 {
305 auto &prev_layer = std::get<0>(_layers[i - 1]);
306 auto &prev_activation = std::get<1>(_layers[i - 1]);
307 prev_layer_output = prev_activation->operator()(prev_layer->output());
308 }
309
310 layer->grad_weights(dz * prev_layer_output.transpose() / _num_examples);
311 layer->grad_weights(layer->grad_weights() +
312 (_regularization_factor / _num_examples) * layer->weights());
313 layer->grad_weights(dz.rowwise().mean());
314
315 da = layer->weights().transpose() * dz;
316 }
